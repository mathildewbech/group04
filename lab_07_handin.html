Making a nice-looking plot

arrow_style <- arrow(
  angle = 20, 
  ends = "first", 
  type = "closed", 
  length = grid::unit(8, "pt")
)

# plot rotation matrix
pca_fit %>%
  tidy(matrix = "rotation") %>%
  pivot_wider(names_from = "PC",
              names_prefix = "PC",
              values_from = "value") %>%
  
  ggplot(aes(PC1, PC2)) +
  geom_segment(xend = 0,
               yend = 0, 
               arrow = arrow_style) +
  geom_text(
    aes(label = column),
    hjust = 1,
    nudge_x = -0.02, 
    color = "#bf8bff"
  ) +
  xlim(-1.25, .5) + 
  ylim(-.5, 1) +
  coord_fixed() + # fixing ratio
  theme_minimal_grid(12)
